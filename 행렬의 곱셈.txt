class Solution {
    public int[][] solution(int[][] arr1, int[][] arr2) {
        int x=arr1.length;
        int y=arr2[0].length;
        int z=arr1[0].length;
        int[][] answer = new int[x][y];
        for(int i=0;i<x;i++){
            for(int j=0;j<y;j++){
                for(int h=0;h<z;h++){
                    answer[i][j]+=arr1[i][h]*arr2[h][j];
                }
            }
        }
        return answer;
    }
}
